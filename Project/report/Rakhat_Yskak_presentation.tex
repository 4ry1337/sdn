\begin{frame}{System Architecture}
\begin{figure}
    \centering
    \includegraphics[width=0.95\textwidth]{figures/architecture.png}
    \caption{Client-server architecture with SSE streaming}
\end{figure}

\begin{block}{Technology Stack}
    \begin{itemize}
        \item \textbf{Frontend:} React + TypeScript + D3.js force-directed graphs
        \item \textbf{Backend:} Next.js with Server-Sent Events (SSE)
        \item \textbf{Controllers:} Floodlight via standard OpenFlow REST API
        \item \textbf{Key Innovation:} Real-time updates without polling from client
    \end{itemize}
\end{block}
\end{frame}

% Slide 4: Key Features (~1 minute)
\begin{frame}{Core Capabilities}

\begin{block}{1. Real-Time Topology Visualization}
\begin{itemize}
    \item D3.js force-directed graph with automatic layout
    \item Interactive: drag nodes, zoom, pan
    \item Color coding: green (controllers), blue (switches), purple (hosts)
\end{itemize}
\end{block}

\begin{block}{2. Traffic Monitoring}
\begin{itemize}
    \item Links colored by utilization: green (low) → yellow (medium) → red (high)
    \item Bottlenecks become immediately visible
    \item Port statistics and bandwidth metrics
\end{itemize}
\end{block}

\begin{block}{3. Multi-Controller Support}
\begin{itemize}
    \item Monitor multiple controllers simultaneously
    \item Health status indicators with auto-reconnect
    \item Failover detection
\end{itemize}
\end{block}

\end{frame}

% Slide 5: Implementation Details (~45 seconds)
\begin{frame}{Technical Implementation}

\begin{block}{Server-Sent Events (SSE) Streaming}
    \begin{itemize}
        \item Backend polls Floodlight REST API (configurable interval)
        \item Server pushes updates to clients via SSE
        \item Client receives real-time topology changes without refresh
        \item Automatic error handling and retry logic
    \end{itemize}
\end{block}

\begin{block}{Flow Statistics \& Events}
    \begin{itemize}
        \item Track flow counts, packet/byte counters per switch
        \item TCAM capacity monitoring with warnings
        \item Timeline of network events: links up/down, switch/host changes
    \end{itemize}
\end{block}

\end{frame}
